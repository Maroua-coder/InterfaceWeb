  <script>
    // Simulation des données (à remplacer par les vraies données plus tard)
    function getRandom(min, max) {
      return (Math.random() * (max - min) + min).toFixed(2);
    }

    function updateValues() {
      document.getElementById("temp").textContent = getRandom(20, 35) + " °C";
      document.getElementById("vib").textContent = getRandom(0.1, 2.5) + " m/s²";
      document.getElementById("sound").textContent = getRandom(30, 90) + " dB";
    }

    // Met à jour toutes les 2 secondes
    setInterval(updateValues, 2000);
  </script>
